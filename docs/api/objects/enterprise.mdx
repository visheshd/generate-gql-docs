An account to manage multiple organizations with consolidated policy and billing.

```graphql
type Enterprise implements Node {
  avatarUrl(size: Int = 0): URI!
  billingInfo: EnterpriseBillingInfo
  createdAt: DateTime!
  databaseId: Int
  description: String
  descriptionHTML: HTML!
  id: ID!
  location: String
  members(
    after: String
    before: String
    deployment: EnterpriseUserDeployment
    first: Int = 0
    last: Int = 0
    orderBy: EnterpriseMemberOrder = "[object Object]"
    organizationLogins: [String!] = []
    query: String
    role: EnterpriseUserAccountMembershipRole
  ): EnterpriseMemberConnection!
  name: String!
  organizations(
    after: String
    before: String
    first: Int = 0
    last: Int = 0
    orderBy: OrganizationOrder = "[object Object]"
    query: String
  ): OrganizationConnection!
  ownerInfo: EnterpriseOwnerInfo
  resourcePath: URI!
  slug: String!
  url: URI!
  userAccounts(
    after: String
    before: String
    first: Int = 0
    last: Int = 0
  ): EnterpriseUserAccountConnection!
  viewerIsAdmin: Boolean!
  websiteUrl: URI
}
```
