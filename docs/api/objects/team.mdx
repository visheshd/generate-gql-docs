A team of users in an organization.

```graphql
type Team implements MemberStatusable, Node, Subscribable {
  ancestors(
    after: String
    before: String
    first: Int = 0
    last: Int = 0
  ): TeamConnection!
  avatarUrl(size: Int = 400): URI
  childTeams(
    after: String
    before: String
    first: Int = 0
    immediateOnly: Boolean = "true"
    last: Int = 0
    orderBy: TeamOrder
    userLogins: [String!] = []
  ): TeamConnection!
  combinedSlug: String!
  createdAt: DateTime!
  databaseId: Int
  description: String
  discussion(number: Int!): TeamDiscussion
  discussions(
    after: String
    before: String
    first: Int = 0
    isPinned: Boolean
    last: Int = 0
    orderBy: TeamDiscussionOrder
  ): TeamDiscussionConnection!
  discussionsResourcePath: URI!
  discussionsUrl: URI!
  editTeamResourcePath: URI!
  editTeamUrl: URI!
  id: ID!
  invitations(
    after: String
    before: String
    first: Int = 0
    last: Int = 0
  ): OrganizationInvitationConnection
  memberStatuses(
    after: String
    before: String
    first: Int = 0
    last: Int = 0
    orderBy: UserStatusOrder = "[object Object]"
  ): UserStatusConnection!
  members(
    after: String
    before: String
    first: Int = 0
    last: Int = 0
    membership: TeamMembershipType = "ALL"
    orderBy: TeamMemberOrder
    query: String
    role: TeamMemberRole
  ): TeamMemberConnection!
  membersResourcePath: URI!
  membersUrl: URI!
  name: String!
  newTeamResourcePath: URI!
  newTeamUrl: URI!
  organization: Organization!
  parentTeam: Team
  privacy: TeamPrivacy!
  repositories(
    after: String
    before: String
    first: Int = 0
    last: Int = 0
    orderBy: TeamRepositoryOrder
    query: String
  ): TeamRepositoryConnection!
  repositoriesResourcePath: URI!
  repositoriesUrl: URI!
  resourcePath: URI!
  slug: String!
  teamsResourcePath: URI!
  teamsUrl: URI!
  updatedAt: DateTime!
  url: URI!
  viewerCanAdminister: Boolean!
  viewerCanSubscribe: Boolean!
  viewerSubscription: SubscriptionState
}
```
