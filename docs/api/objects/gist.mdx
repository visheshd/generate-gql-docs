A Gist.

```graphql
type Gist implements Node, Starrable, UniformResourceLocatable {
  comments(
    after: String
    before: String
    first: Int = 0
    last: Int = 0
  ): GistCommentConnection!
  createdAt: DateTime!
  description: String
  files(limit: Int = 10, oid: GitObjectID): [GistFile]
  forks(
    after: String
    before: String
    first: Int = 0
    last: Int = 0
    orderBy: GistOrder
  ): GistConnection!
  id: ID!
  isFork: Boolean!
  isPublic: Boolean!
  name: String!
  owner: RepositoryOwner
  pushedAt: DateTime
  resourcePath: URI!
  stargazerCount: Int!
  stargazers(
    after: String
    before: String
    first: Int = 0
    last: Int = 0
    orderBy: StarOrder
  ): StargazerConnection!
  updatedAt: DateTime!
  url: URI!
  viewerHasStarred: Boolean!
}
```
