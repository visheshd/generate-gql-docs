A workflow run.

```graphql
type WorkflowRun implements Node {
  checkSuite: CheckSuite!
  createdAt: DateTime!
  databaseId: Int
  deploymentReviews(
    after: String
    before: String
    first: Int = 0
    last: Int = 0
  ): DeploymentReviewConnection!
  id: ID!
  pendingDeploymentRequests(
    after: String
    before: String
    first: Int = 0
    last: Int = 0
  ): DeploymentRequestConnection!
  resourcePath: URI!
  runNumber: Int!
  updatedAt: DateTime!

}
```
